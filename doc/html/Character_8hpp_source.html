184  bool isCollision(std::shared_ptr<bmb::block::Block> &block) {
185  if (!_alive)
186  return false;
187  if (block->type() == bmb::block::EMPTY && block->isBonus() && isCollisionB(block)) {
188  auto bonus = block->getPowerUp();
189  if (bonus->tag() == bmb::items::SPEED)
190  _speedBonus += bonus->getBonus();
191  if (bonus->tag() == bmb::items::RANGE)
192  _rangeBomb += static_cast<int>(bonus->getBonus());
193  if (bonus->tag() == bmb::items::BOMB)
194  _listBomb.push_back(std::shared_ptr<bmb::items::Bomb>(new bmb::items::Bomb()));
195  }
196  if (block->type() == bmb::block::EMPTY || block->type() == bmb::block::SPAWN)
197  return false;
198  BoundingBox box1 = block->getModel().getBoundingBox();
199  BoundingBox box2 = getModel().getBoundingBox();
200 
201  if (CheckCollisionBoxes(box1, box2)) {
202  _model.setPosition(_oldPosition);
203  return true;
204  }
205  return false;
206  }
